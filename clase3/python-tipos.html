📦 Tipos de Datos

Selecciona un tipo de dato para ver su código y visualización en memoria.

tipos_int.py 
# Tipo: Entero (int)
# Números sin decimales. Positivos, negativos o cero.

edad        = 22
temperatura = -5
año         = 2026
cero        = 0

# Operaciones comunes con enteros
suma    = edad + 10         # 32
doble   = edad * 2          # 44
entero  = 17 // 5          # 3  (división entera)
residuo = 17 %  5          # 2  (módulo)
potencia= 2  ** 8          # 256

print(edad, type(edad))  # 22 <class 'int'>
# Tipo: Flotante (float)
# Números con decimales. Alta precisión.

precio   = 9.99
pi       = 3.14159
negativo = -0.5
notacion = 1.5e3    # 1500.0 (notación científica)

# División normal siempre retorna float
resultado = 10 / 3     # 3.3333...

# Conversión entre tipos
num_int   = int(3.7)    # 3  (trunca, no redondea)
num_round = round(3.7) # 4  (redondeo estándar)

print(precio, type(precio)) # 9.99 <class 'float'>
# Tipo: Cadena (str)
# Texto. Secuencia inmutable de caracteres.

nombre  = "Germán"
curso   = 'TI3011'
multiline= """Hola,
Mundo."""

# F-strings (interpolación moderna)
saludo  = f"Hola, {nombre}!"  # "Hola, Germán!"

# Métodos de cadena
mayus   = nombre.upper()      # "GERMÁN"
longitud= len(nombre)         # 6
contiene= "á" in nombre      # True
trozo   = nombre[0:3]         # "Ger" (slicing)

print(type(nombre))  # <class 'str'>
# Tipo: Booleano (bool)
# Solo dos valores: True o False.

es_mayor  = True
tiene_saldo = False

# Resultado de operaciones de comparación
mayor_edad = 22 >= 18   # True
iguales    = 5  == 5    # True
distintos  = 3  != 4    # True

# Booleanos en condicionales
if es_mayor and tiene_saldo:
    print("Puede comprar")
else:
    print("No puede comprar")

# int(True)=1, int(False)=0
print(type(es_mayor))  # <class 'bool'>
# Función type() — inspecciona el tipo

a = 42
b = 3.14
c = "hola"
d = True

print(type(a))  # <class 'int'>
print(type(b))  # <class 'float'>
print(type(c))  # <class 'str'>
print(type(d))  # <class 'bool'>

# Conversión de tipos (casting)
print(int("100"))    # 100 (str → int)
print(float(5))      # 5.0 (int → float)
print(str(3.14))     # "3.14" (float → str)
print(bool(0))       # False (0 = False)
print(bool(42))      # True (≠0 = True)
