🗂️ Estructuras de Datos

Colecciones que permiten almacenar y organizar varios valores en una sola variable. Selecciona una para ver su sintaxis y comportamiento.

listas.py 
# Listas — colección ORDENADA y MUTABLE
# Permite duplicados. Se accede por índice.

frutas = ["manzana", "pera", "uva", "kiwi"]

# Acceso por índice (empieza en 0)
print(frutas[0])      # "manzana"
print(frutas[-1])     # "kiwi" (desde el final)
print(frutas[1:3])    # ["pera", "uva"] (slicing)

# Modificar (son mutables)
frutas[0] = "naranja"

# Métodos comunes
frutas.append("mango")     # añade al final
frutas.insert(1, "banana") # inserta en pos. 1
frutas.remove("uva")       # elimina por valor
ultima = frutas.pop()      # saca y devuelve la última
frutas.sort()                # ordena in-place

len(frutas)                  # cantidad de elementos
"pera" in frutas            # True / False
# Diccionarios — pares CLAVE: VALOR
# Acceso por clave, no por índice. Mutables.

estudiante = {
    "nombre": "Ana",
    "edad":   21,
    "curso":  "TI3011",
    "notas":  [6.5, 5.8, 7.0]
}

# Acceso por clave
print(estudiante["nombre"])   # "Ana"
print(estudiante.get("email", "-"))  # "-" si no existe

# Modificar / agregar
estudiante["edad"] = 22          # actualiza
estudiante["email"] = "ana@inacap.cl"  # añade

# Eliminar
del estudiante["curso"]
estudiante.pop("edad", None)

# Recorrer claves, valores o ambos
for clave, valor in estudiante.items():
    print(clave, "→", valor)

estudiante.keys()    # dict_keys([...])
estudiante.values()  # dict_values([...])
# Tuplas — colección ORDENADA pero INMUTABLE
# Útiles para datos que NO deben cambiar.

punto     = (10, 20)
colores   = ("rojo", "verde", "azul")
una_sola  = (42,)            # ¡coma obligatoria!

# Acceso por índice (igual que listas)
print(punto[0])           # 10
print(colores[-1])        # "azul"

# NO se puede modificar
# punto[0] = 5  → TypeError

# Desempaquetado (unpacking)
x, y = punto
r, g, b = colores

# Caso típico: retornar varios valores
def dividir(a, b):
    return a // b, a % b

cociente, resto = dividir(17, 5)  # (3, 2)

tuple

( ) · ordenada · inmutable

Igual que una lista, pero NO se puede modificar tras crearla. Más liviana y segura para datos fijos (coordenadas, fechas, configuración).
