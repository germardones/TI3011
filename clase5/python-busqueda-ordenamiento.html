# ── BÚSQUEDA LINEAL ─────────────────────────────
# Recorre cada elemento de izquierda a derecha
# comparando con el objetivo. Para cuando lo encuentra.

def busqueda_lineal(lista, objetivo):
    for i, elemento in enumerate(lista):
        if elemento == objetivo:
            return i   # índice del primer encontrado
    return -1        # -1 si no existe

numeros = [34, 7, 23, 32, 5, 62]
print(busqueda_lineal(numeros, 23))  # → 2
print(busqueda_lineal(numeros, 99))  # → -1

# ── VARIANTE: buscar TODOS los índices ───────────
def buscar_todos(lista, objetivo):
    return [i for i, x in enumerate(lista) if x == objetivo]

datos = [3, 7, 3, 9, 3]
print(buscar_todos(datos, 3))  # → [0, 2, 4]

# ── VARIANTE: buscar en lista de objetos ─────────
alumnos = [
    {"nombre": "Ana",  "nota": 6.5},
    {"nombre": "Luis", "nota": 4.0},
    {"nombre": "Sofía","nota": 7.0},
]

def buscar_alumno(lista, nombre):
    for alumno in lista:
        if alumno["nombre"] == nombre:
            return alumno
    return None

print(buscar_alumno(alumnos, "Sofía"))
# → {"nombre": "Sofía", "nota": 7.0}

# ── VARIANTE: búsqueda con condición ─────────────
def primer_aprobado(alumnos):
    for a in alumnos:
        if a["nota"] >= 4.0:
            return a["nombre"]
    return "Nadie aprobó"

# ¿Cuándo usar lineal?
# ✅ Lista NO ordenada
# ✅ Lista pequeña (< 500 elementos)
# ✅ Buscar por condición compleja (no solo igualdad)
# ❌ Lista grande y ordenada → usar binaria
# ── BÚSQUEDA BINARIA ─────────────────────────────
# REQUIERE lista ORDENADA. En cada paso descarta
# la mitad de los elementos restantes.

def busqueda_binaria(lista, objetivo):
    izq = 0                   # límite izquierdo
    der = len(lista) - 1    # límite derecho

    while izq <= der:        # mientras haya rango
        medio = (izq + der) // 2  # punto central

        if lista[medio] == objetivo:
            return medio               # ¡encontrado!
        elif lista[medio] < objetivo:
            izq = medio + 1           # buscar en mitad derecha
        else:
            der = medio - 1           # buscar en mitad izquierda

    return -1                    # no existe en la lista

# ── Ejemplo paso a paso: buscar 32 ───────────────
lista = [5, 7, 23, 32, 34, 62]

# Paso 1: izq=0 der=5 → medio=2 → lista[2]=23 < 32
#         23 < 32, entonces izq = 2+1 = 3
# Paso 2: izq=3 der=5 → medio=4 → lista[4]=34 > 32
#         34 > 32, entonces der = 4-1 = 3
# Paso 3: izq=3 der=3 → medio=3 → lista[3]=32 == 32 ✓

print(busqueda_binaria(lista, 32))   # → 3
print(busqueda_binaria(lista, 10))   # → -1 (no existe)

import bisect

if i < len(lista) and lista[i] == 32:
    print(f"Encontrado en índice {i}")

# ✅ Lista ORDENADA (condición obligatoria)

# ✅ Datos que no deben cambiar
# ✅ Claves de diccionarios
# ❌ No puedes usar .append() ni .remove()
# Diccionario — pares clave:valor, mutable
alumno = {
    "nombre": "Luis",
    "edad": 20,
    "notas": [6.0, 5.5, 7.0]
}

# Acceso por clave
print(alumno["nombre"])          # "Luis"
print(alumno.get("edad", 0))     # 20 (con valor por defecto)

# Recorrido — solo claves
for clave in alumno:
    print(clave)

# Recorrido — claves y valores
for clave, valor in alumno.items():
    print(f"{clave}: {valor}")

# Recorrido — solo valores
for valor in alumno.values():
    print(valor)

# Agregar / modificar
alumno["carrera"] = "Civil"
alumno["edad"] = 21
# Set (conjunto) — sin duplicados, sin orden
colores = {"rojo", "verde", "azul", "rojo"}
print(colores)  # {"rojo", "verde", "azul"} — solo 3

# Recorrido (sin garantía de orden)
for color in colores:
    print(color)

# Operaciones de conjuntos
a = {1, 2, 3, 4}
b = {3, 4, 5, 6}

union        = a | b   # {1, 2, 3, 4, 5, 6}
interseccion = a & b   # {3, 4}
diferencia   = a - b   # {1, 2}

# Verificar pertenencia — O(1), más rápido que lista

# Eliminar duplicados de una lista
lista = [1, 2, 2, 3, 3, 3]
unicos = list(set(lista))  # [1, 2, 3]

Búsqueda Lineal

O(n) · sin requisitos · simple

Recorre cada elemento uno por uno hasta encontrar el objetivo. No requiere que la lista esté ordenada.

O(n)
Complejidad lineal
En el peor caso, recorre todos los n elementos. Con 1000 elementos: hasta 1000 comparaciones.

Búsqueda Binaria

O(log n) · lista ordenada · divide y vencerás

Divide el espacio de búsqueda a la mitad en cada paso. Mucho más eficiente que la lineal, pero requiere lista ordenada.

O(log n)
Complejidad logarítmica
Con 1000 elementos: máximo ~10 comparaciones. Con 1.000.000: ~20 comparaciones.
